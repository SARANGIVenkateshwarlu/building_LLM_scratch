## 3. Weight Assignment: `load_weights_into_gpt()`

**Core function**: Map `params` → `model` layers.

### Step 3.1: Helper `assign()` Function

```python
def assign(left, right):
    """Safe weight assignment with shape check"""
    if left.shape != right.shape:
        raise ValueError(f"Shape mismatch: {left.shape} vs {right.shape}")
    left.data = right
    return left
```


### Step 3.2: Full Weight Loading

```python
def load_weights_into_gpt(model, params):
    # 1. Token + Positional Embeddings
    assign(model.token_embedding.weight, torch.tensor(params['wte']))
    assign(model.position_embedding.weight, torch.tensor(params['wpe']))
    
    # 2. Transformer Blocks (12 layers)
    for i in range(model.config.n_layers):
        block = model.transformer.blocks[i]
        
        # Attention: Split fused QKV  → Q + KV
        fused_qkv_w = torch.tensor(params['blocks'][i]['attn']['c_attn.w'])
        fused_qkv_b = torch.tensor(params['blocks'][i]['attn']['c_attn.b'])
        
        block.attention.W_Q.weight.data = fused_qkv_w[:768,  :768]   # Q
        block.attention.W_KV.weight.data = fused_qkv_w[768:,  :1536] # KV
        block.attention.W_Q.bias.data   = fused_qkv_b[:768]         # Q bias  
        block.attention.W_KV.bias.data  = fused_qkv_b[768:]         # KV bias
        
        # Attention output projection
        assign(block.attention.W_O.weight, torch.tensor(params['blocks'][i]['attn']['c_proj.w']))
        assign(block.attention.W_O.bias,   torch.tensor(params['blocks'][i]['attn']['c_proj.b']))
        
        # FFN: fc1 (expand 768→3072), fc2 (project 3072→768)
        assign(block.feed_forward.fc1.weight, torch.tensor(params['blocks'][i]['ffn']['c_fc.w']))
        assign(block.feed_forward.fc1.bias,   torch.tensor(params['blocks'][i]['ffn']['c_fc.b']))
        assign(block.feed_forward.fc2.weight, torch.tensor(params['blocks'][i]['ffn']['c_proj.w']))
        assign(block.feed_forward.fc2.bias,   torch.tensor(params['blocks'][i]['ffn']['c_proj.b']))
        
        # LayerNorms (scale + shift)
        assign(block.ln1.gamma, torch.tensor(params['blocks'][i]['ln_1']['g']))
        assign(block.ln1.beta,  torch.tensor(params['blocks'][i]['ln_1']['b']))
        assign(block.ln2.gamma, torch.tensor(params['blocks'][i]['ln_2']['g']))
        assign(block.ln2.beta,  torch.tensor(params['blocks'][i]['ln_2']['b']))
    
    # 3. Final LayerNorm
    assign(model.ln_f.gamma, torch.tensor(params['ln_f.g']))
    assign(model.ln_f.beta,  torch.tensor(params['ln_f.b']))
    
    # 4. **Weight Tying**: LM Head = Token Embeddings (saves 38M params!)
    model.lm_head.weight = model.token_embedding.weight
    
    print("✅ All GPT-2 weights loaded successfully!")

# Execute:
load_weights_into_gpt(model, params)
model.eval()
```
